

# 根目录
cd D:\sichaun-travel-system

# 创建前端项目
# npm create vite@latest client -- --template vue
cd D:\sichaun-travel-system\client
npm install
npm install vue-router@4 pinia axios element-plus @amap/amap-jsapi-loader
//npm install element-plus @element-plus/icons-vue
cd ..

# 创建后端项目
cd D:\sichaun-travel-system\server
npm init -y
npm install express cors mysql2 bcryptjs jsonwebtoken axios cheerio node-cron
npm install nodemon --save-dev
cd ..


# 登录 MySQL
mysql -u root -p
mysql8
# 执行初始化脚本
source database/init.sql;
EXIT

# 终端 1 — 启动后端
cd server
npx nodemon app.js
# 服务运行于 http://localhost:3000

# 终端 2 — 启动前端
cd D:\sichaun-travel-system\client
npm run dev
# 访问 http://localhost:5173

# 终端 3 — 启动爬虫定时任务（可选）
cd server
node -e "require('./crawler/scheduler').start()"
